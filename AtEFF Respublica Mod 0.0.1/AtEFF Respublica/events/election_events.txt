# Annual
character_event = {
	id = laws.4000
	desc = "EVTDESCLAWS4000"
	picture = GFX_evt_election
	
	only_rulers = yes
	
	trigger = {
		NOT = { has_character_modifier = constitutional_crisis }
		any_demesne_title = {
			OR = {
				has_law = crown_term_length_0
				has_law = local_term_length_0
			}
			OR = {
				has_law = succ_democratic_elective
			}
			OR = {
				NOT = { has_title_flag = flag_election_cycle }
				had_title_flag = { flag = flag_election_cycle days = 365 }
			}
			NOT = { has_title_flag = flag_holding_election }
		}
	}
	
	option = {
		trigger = { NOT = { is_primary_war_defender = yes } }
		name = "EVTOPTACCEPT"
		any_demesne_title = {
			limit = {
				OR = {
					has_law = crown_term_length_0
					has_law = local_term_length_0
				}
				OR = {
					has_law = succ_democratic_elective
				}
				OR = {
					NOT = { has_title_flag = flag_election_cycle }
					had_title_flag = { flag = flag_election_cycle days = 365 }
				}
				NOT = { has_title_flag = flag_holding_election }
			}
			set_title_flag = flag_holding_election
		}
		character_event = { id = laws.5000 }
		ai_chance = { factor = 100 }
	}
	
	option = {
		name = "EVTOPTCLAWS4000" # Can't change horses in midstream
		trigger = { is_primary_war_defender = yes }
		hidden_tooltip = {
			any_demesne_title = {
				limit = {
					OR = {
						has_law = crown_term_length_0
						has_law = local_term_length_0
					}
					OR = {
						has_law = succ_democratic_elective

						NOT = { has_title_flag = flag_election_cycle }

					NOT = { has_title_flag = flag_holding_election }
				}
				clr_title_flag = flag_election_cycle
				set_title_flag = flag_election_cycle
			}

# You have agreed to hold elections!
character_event = {
	id = laws.5000
	desc = "EVTDESCLAWS5000"
	picture = GFX_evt_election
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTCONFIRM"
		any_independent_ruler = { character_event = { id = laws.5001 } }
		hidden_tooltip = {
			any_demesne_title = {
				limit = {
					has_title_flag = flag_holding_election
					has_law = succ_patrician_elective
					is_primary_holder_title = no
				}
				clr_title_flag = flag_holding_election
			}
		}
		# Federal Elections
		any_demesne_title = {
			limit = {
				has_title_flag = flag_holding_election
				OR = {
					has_law = succ_feudal_elective
					has_law = succ_elective_gavelkind
					has_law = succ_tanistry
				}
				current_heir = { NOT = { has_character_flag = flag_already_chosen_for_federal_election } }
			}
			current_heir = {
				clr_character_flag = flag_already_chosen_for_federal_election
				set_character_flag = flag_already_chosen_for_federal_election
				character_event = { id = laws.5100 tooltip = "EVTTOOLTIPLAWS5100" days = 1 }
			}
		}
		# Popular Elections
		any_demesne_title = {
			limit = {
				has_title_flag = flag_holding_election
				OR = {
					has_law = succ_democratic_elective
				}
				OR = {
					has_law = succ_democratic_elective
					is_primary_holder_title = yes
				}
				current_heir = { NOT = { has_character_flag = flag_already_chosen_for_popular_election } }
			}
			current_heir = {
				clr_character_flag = flag_already_chosen_for_popular_election
				set_character_flag = flag_already_chosen_for_popular_election
				character_event = { id = laws.5200 tooltip = "EVTTOOLTIPLAWS5200" days = 1 }
			}
		}
	}
}

# A challenger has beaten you in a federal election!
character_event = {
	id = laws.5110
	desc = "EVTDESCLAWS5110"
	picture = GFX_evt_election
	is_triggered_only = yes
	
	trigger = { FROM = { check_variable = { which = "victory_margin_federal" value = 1 } } }
	
	option = {
		name = "EVTOPTCONFIRM"
		any_demesne_title = {
			limit = {
				has_title_flag = flag_holding_election
				has_law = succ_feudal_elective
				current_heir = { character = FROM }
			}
			# If there's a designated capital, give it over as well
			ROOT = {
				random_demesne_province = {
					limit = {
						owner = { character = ROOT }
						empire = { title = PREVPREVPREV }
						has_province_modifier = empire_capital_system
					}
					usurp_title_plus_barony_if_unlanded_and_vassals  = FROM
				}
				random_demesne_province = {
					limit = {
						owner = { character = ROOT }
						kingdom = { title = PREVPREVPREV }
						has_province_modifier = kingdom_capital_system
					}
					usurp_title_plus_barony_if_unlanded_and_vassals  = FROM
				}
			}
			set_title_flag = flag_just_won_federal_election
			usurp_title_plus_barony_if_unlanded_and_vassals  = FROM
		}
	}
}

# Called from any titles you win in a federal election
character_event = {
	id = laws.5115
	desc = "EVTDESCLAWS5115"
	picture = GFX_evt_election
	is_triggered_only = yes
	
	trigger = { FROM = { has_title_flag = flag_just_won_federal_election } }
	
	option = {
		name = "EVTOPTCONFIRM"
		FROM = {
			# End election
			clr_title_flag = flag_just_won_federal_election
			clr_title_flag = flag_holding_election
		}
		# Gain prestige equivalent to gained titles
		if = {
			limit = { FROM = { tier = count } }
			prestige = 25
		}
		if = {
			limit = { FROM = { tier = duke } }
			prestige = 50
		}
		if = {
			limit = { FROM = { tier = king } }
			prestige = 75
		}
		if = {
			limit = { FROM = { tier = emperor } }
			prestige = 100
		}
		# Inform all relevant personnel of the new incumbent
		hidden_tooltip = {
			any_realm_character = {
				limit = {
					NOT = { character = ROOT }
					OR = {
						any_liege = { character = ROOT }
						any_vassal = { character = ROOT }
					}
				}
				character_event = { id = laws.5116 }
			}
		}
	}
}

# A challenger has lost to you in a federal election!
character_event = {
	id = laws.5120
	desc = "EVTDESCLAWS5120"
	picture = GFX_evt_election
	is_triggered_only = yes
	
	trigger = { FROM = { NOT = { check_variable = { which = "victory_margin_federal" value = 1 } } } }
	
	option = {
		name = "EVTOPTCONFIRM"
		any_demesne_title = {
			limit = {
				has_title_flag = flag_holding_election
				OR = {
					has_law = succ_feudal_elective
					has_law = succ_tanistry
				}
				current_heir = { character = FROM }
			}
			
			# End the election and reset the election cycle
			clr_title_flag = flag_holding_election
			clr_title_flag = flag_election_cycle
			set_title_flag = flag_election_cycle
			
			# Gain prestige based on title rank
			if = {
				limit = { tier = count }
				ROOT = { prestige = 25 }
			}
			if = {
				limit = { tier = duke }
				ROOT = { prestige = 50 }
			}
			if = {
				limit = { tier = king }
				ROOT = { prestige = 75 }
			}
			if = {
				limit = { tier = emperor }
				ROOT = { prestige = 100 }
			}
			hidden_tooltip = {
				current_heir = {
					if = {
						limit = { NOT = { has_character_flag = flag_informed_of_election_loss } }
						set_character_flag = flag_informed_of_election_loss
						character_event = { id = laws.5125 }
					}
				}
			}
		}
		hidden_tooltip = {
			ROOT = {
				any_realm_character = {
					limit = {
						NOT = { character = ROOT }
						OR = {
							any_liege = { character = ROOT }
							any_vassal = { character = ROOT }
						}
					}
				}
			}
		}
	}
}
